## copy sh

```sh
#/bin/bash

dst_date=`date "+%Y%m%d"`
dst=${dst_date}.md

[ -e "${dst}" ] && echo "exist file ${dst}" && code ${dst} && exit

d=0

while [ ! -e "${src}" ]
do

  d=$(( d + 1 ))

  src_date=`date -v -${d}d "+%Y%m%d"`
  src=${src_date}.md

done

[ ! -e "${src}" ] && echo "not exist file ${src}" && exit

echo "copy ${src} -> ${dst}"

cp "${src}" "${dst}"

# 1行目の日付を変更
code ${dst}

```
